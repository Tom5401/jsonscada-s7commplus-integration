**buildTree(docs, dbName) function:**
```javascript
function buildTree(docs, dbName) {
  const root = { id: dbName, name: dbName, isLeaf: false, children: [] }
  for (const doc of docs) {
    const browsePath = doc.protocolSourceBrowsePath
    if (!browsePath) continue
    const segments = browsePath.split('.')
    const remaining = segments.slice(1)  // skip DB name (segments[0])
    if (remaining.length === 0) continue
    let current = root
    for (let i = 0; i < remaining.length; i++) {
      const seg = remaining[i]
      const fullId = segments.slice(0, i + 2).join('.')
      const isLast = i === remaining.length - 1
      let child = current.children.find(c => c.id === fullId)
      if (!child) {
        child = {
          id: fullId,
          name: seg,
          isLeaf: isLast,
          children: isLast ? undefined : [],
          ...(isLast ? {
            type: doc.type || '',
            value: doc.value,
            address: doc.protocolSourceObjectAddress || '',
          } : {})
        }
        current.children.push(child)
      }
      current = child
    }
  }
  return root
}
```

IMPORTANT: Leaf nodes MUST have `children: undefined` (not `children: []`) to prevent v-treeview from rendering an expand arrow on leaf nodes. Non-leaf nodes use `children: []` or populated arrays.

**refreshValues function (per D-05: in-place value update preserving expand state):**
1. `const res = await fetch('/Invoke/auth/listS7PlusTagsForDb?connectionNumber=' + connectionNumber.value + '&dbName=' + encodeURIComponent(dbName.value))`
2. `const docs = await res.json()`
3. Build a Map: `const docMap = new Map(docs.map(d => [d.protocolSourceBrowsePath, d]))`
4. Walk the existing tree in `treeItems.value[0]` recursively. For each leaf node (`node.isLeaf === true`), look up `docMap.get(node.id)` and if found, mutate `node.value = doc.value` in-place. Do NOT replace treeItems array reference.
5. After patching values, call `touchExpandedLeafTags()` to keep TTL alive.

**patchLeafValues(treeRoot, freshDocs) helper:**
```javascript
function patchLeafValues(treeRoot, freshDocs) {
  const docMap = new Map(freshDocs.map(d => [d.protocolSourceBrowsePath, d]))
  function walk(node) {
    if (node.isLeaf) {
      const doc = docMap.get(node.id)
      if (doc) node.value = doc.value
    } else if (node.children) {
      for (const child of node.children) walk(child)
    }
  }
  walk(treeRoot)
}
```

**touchExpandedLeafTags function (per TAGTREE-02: touch on expand + refresh):**
1. Collect all leaf tags that are currently visible (under an opened ancestor node).
2. Use `openedNodes.value` as the set of opened node IDs.
3. Walk the tree: for each node, if its id is in openedNodes (or it is root which is always open), recurse into children. For leaf nodes under an opened parent, collect `{ connectionNumber: connectionNumber.value, protocolSourceObjectAddress: node.address }`.
4. Guard: if collected array is empty, return without calling API (backend rejects empty array with 400 — Pitfall 6).
5. `await fetch('/Invoke/auth/touchS7PlusActiveTagRequests', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(expandedLeafs) })`

**getExpandedLeafTags(treeRoot, openedIds) helper:**
```javascript
function getExpandedLeafTags(treeRoot, openedIds) {
  const openSet = new Set(openedIds)
  const result = []
  function walk(node, parentOpen) {
    const isOpen = parentOpen || openSet.has(node.id)
    if (node.isLeaf && parentOpen) {
      result.push({
        connectionNumber: connectionNumber.value,
        protocolSourceObjectAddress: node.address,
      })
    } else if (node.children) {
      for (const child of node.children) walk(child, isOpen)
    }
  }
  if (treeRoot) {
    for (const child of treeRoot.children || []) walk(child, openSet.has(treeRoot.id))
  }
  return result
}
```

**refreshValues orchestration:**
```javascript
const refreshValues = async () => {
  if (!dbName.value || !connectionNumber.value) return
  try {
    const res = await fetch(
      `/Invoke/auth/listS7PlusTagsForDb?connectionNumber=${connectionNumber.value}&dbName=${encodeURIComponent(dbName.value)}`
    )
    const docs = await res.json()
    if (Array.isArray(docs) && treeItems.value[0]) {
      patchLeafValues(treeItems.value[0], docs)
      await touchExpandedLeafTags()
    }
  } catch (err) {
    console.warn('Failed to refresh tag values:', err)
  }
}
```

**touchExpandedLeafTags orchestration:**
```javascript
const touchExpandedLeafTags = async () => {
  if (!treeItems.value[0]) return
  const tags = getExpandedLeafTags(treeItems.value[0], openedNodes.value)
  if (tags.length === 0) return  // Pitfall 6: backend rejects empty array
  try {
    await fetch('/Invoke/auth/touchS7PlusActiveTagRequests', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(tags),
    })
  } catch (err) {
    console.warn('Failed to touch active tag requests:', err)
  }
}
```
